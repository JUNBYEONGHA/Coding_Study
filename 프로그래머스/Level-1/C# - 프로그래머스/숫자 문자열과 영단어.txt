using System;

public class Solution {
    public int solution(string s) {
        int answer = 0;
        s = s.Replace("zero","0");
        s = s.Replace("one","1");
        s = s.Replace("two","2");
        s = s.Replace("three","3");
        s = s.Replace("four","4");
        s = s.Replace("five","5");
        s = s.Replace("six","6");
        s = s.Replace("seven","7");
        s = s.Replace("eight","8");
        s = s.Replace("nine","9");
        answer = Convert.ToInt32(s);
        return answer;
    }
}
