#include <string>
#include <vector>

using namespace std;
#define INF 1000000007

int dp[600001];

int solution(int n) {
    int answer = 0;
    dp[1] = 1;
    dp[2] = 2;
    for(int i = 3 ; i <= n ; i++)
    {
        dp[i] = (dp[i-1] + dp[i-2]) % INF;
    }
    answer = dp[n];
    return answer;
}
